//Create a table 

CREATE TABLE posts (id INT PRIMARY KEY AUTO_INCREMENT, title TEXT NOT NULL, description LONGTEXT NOT NULL, image TEXT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP); 

  CREATE TABLE users (id INT PRIMARY KEY AUTO_INCREMENT , name VARCHAR(191) NOT NULL, email VARCHAR(191) NOT NULL UNIQUE, password VARCHAR(191) NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);
